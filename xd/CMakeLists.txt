SET(XDBASEDIR .)
FILE(GLOB_RECURSE SRC_BASE_XD ${XDBASEDIR}/base/*.h ${XDBASEDIR}/base/*.cpp)
FILE(GLOB_RECURSE SRC_NET_XD ${XDBASEDIR}/net/*.h ${XDBASEDIR}/net/*.cpp)
SOURCE_GROUP("base" FILES ${SRC_BASE_XD})
SOURCE_GROUP("net" FILES ${SRC_NET_XD})
SET(ALL_SOURCE ${SRC_BASE_XD}
               ${SRC_NET_XD})
ADD_LIBRARY(xd ${ALL_SOURCE})
TARGET_LINK_LIBRARIES(xd pthread)
